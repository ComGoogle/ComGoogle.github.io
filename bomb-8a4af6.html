实验三bomb

对bomb进行分析，找到6个拆弹密码

输错密码就会

在ubuntu14.05用objdump输出了两个反汇编文件，其实只有一个有用，bomb_dis.txt，不清楚bomb-quiet是干啥的

实在是不想用虚拟机的GDB，在wsl上安装32位支持

阶段一 比较

查看阶段1的代码发现他调用了一个strings_not_equal 8048b62: e8 7d 04 00 00 call 8048fe4 <strings_not_equal>

查看这个函数，发现里面在做字符串比较，而且会先计算长度以及给出了字符串的地址（一共两个存在ebx和esi），输入一个单字符串测试，发现ebx的字符串长度为1，esi的字符串长度为39，可知第一个密码长度为39字节，首地址在0x804a1c4

Brownie, you are doing a heck of a job.

后知后觉好像知道了首地址的时候就可以x/s了，不需要去管长度

也就是说其实 一找到phase_1，就发现0x804a1c4

#include <stdio.h>

int main() {
    int array[16] = {10, 2, 14, 7, 8, 12, 15, 11, 0, 4, 1, 13, 3, 9, 6, 5};
    int ecx = 0; // 用于累加数组元素的值
    int edx = 0; // 用于记录循环次数
    int eax = 0; // 初始索引值
    
    scanf("%d", &eax);
    // 循环，直到数组元素的值为15
    do {
        edx++; // 每次循环递增计数器
        ecx += array[eax]; // 累加当前索引处的数组元素值
        eax = array[eax]; // 更新索引为当前元素的值
    } while (eax != 15);

    printf("循环了 %d 次，元素之和为 %d\n", edx, ecx);

    return 0;
}

struct TreeNode {
    int value;
    struct TreeNode *left;

int fun7(struct TreeNode *node, int a) {
    if (node == NULL) {
        return -1;  // 与 0xffffffff 相对应，可能用于表示未找到或错误
    }

    int nodeValue = node->value;
    if (a < nodeValue) {
        // 搜索左子树
        int result = fun7(node->left, a);
        return 2 * result;  // 将结果翻倍
    } else if (a > nodeValue) {
        // 搜索右子树
        int result = fun7(node->right, a);
        return 2 * result + 1;  // 将结果翻倍并加1
    } else {
        // 找到值，返回0
        return 0;
    }
}
