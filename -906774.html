void myKey_callback()
{ char k1,k3;
	k1=GetKeyAct(enumKey1);
	k3=GetKeyAct(enumKey3);
	if( k1 == enumKeyPress ) Seg7Print(10,10,10,10,10,10,10,18); //按下为H
	  else if( k1 == enumKeyRelease ) Seg7Print(10,10,10,10,10,10,10,10);  //抬起为灭
	if( k3 == enumKeyPress ) LedPrint(118); //按下为H
	  else if( k3 == enumKeyRelease ) LedPrint(0);  //抬起为灭
}

void main(){
	DisplayerInit();   
	KeyInit(); 
	SetDisplayerArea(0,7);
	SetEventCallBack(enumEventKey, myKey_callback); //按键事件发生，初始化为灭
	MySTC_Init();
	while(1)
		MySTC_OS();
}

bit flag = 0;
unsigned long time=0;
unsigned int min=0,sec=0,ms=0;
char mins[2]={0},secs[2]={0},mss[2]={0};

void my10ms_callback(){
	if(flag==1) time++;
	if(time>=594000) time=0,flag=0; //超过99分钟归零与停止
	
	ms=time%100;  //ms当前值为对100取模
	sec=(time/100)%60; //time/100得到秒，对分钟单位60s取模
	min=time/6000; //一分钟有6000个time
	
	mss[0]=ms%10;
	mss[1]=(ms/10)%10;
	secs[0]=sec%10;
	secs[1]=(sec/10)%10;
	mins[0]=min%10;
	mins[1]=(min/10)%10;
	
	Seg7Print(mins[1],mins[0],40,secs[1],secs[0],40,mss[1],mss[0]);
}

void myKey_callback()
{ char k1,k3;
	k1=GetKeyAct(enumKey1);
	k3=GetKeyAct(enumKey3);
	if( k1 == enumKeyPress ) //按下k1控制开始与暂停
	{
		flag=~flag;
		SetBeep(5000,5);
	}
	if( k3 == enumKeyPress )  //按下k3清空秒表
	{
		flag=0;
		time=0;
		SetBeep(1000,50);//发声1KHz,0.5s
	}
 
}

int main(){
	DisplayerInit(); 	
	BeepInit();
	KeyInit();
	SetDisplayerArea(0,7);
	LedPrint(0);
	
	SetEventCallBack(enumEventKey, myKey_callback);
	SetEventCallBack(enumEventSys10mS, my10ms_callback);
	
	MySTC_Init();
	while(1)
		MySTC_OS();
}
